Playwright → claude.ai — рабочая схема:

python
from playwright.async_api import async_playwright
import asyncio

async def ask_claude(prompt: str) -> str:
    async with async_playwright() as p:
        browser = await p.chromium.launch(headless=True)
        page = await browser.new_page()
        
        # Открываем чат
        await page.goto("https://claude.ai/new")
        await page.wait_for_selector('[contenteditable="true"]')
        
        # Вводим промпт
        await page.fill('[contenteditable="true"]', prompt)
        await page.keyboard.press("Enter")
        
        # Ждём ответа
        await page.wait_for_selector('.assistant-message', timeout=60000)
        await asyncio.sleep(3)  # ждём завершения стриминга
        
        response = await page.inner_text('.assistant-message:last-child')
        await browser.close()
        return response
