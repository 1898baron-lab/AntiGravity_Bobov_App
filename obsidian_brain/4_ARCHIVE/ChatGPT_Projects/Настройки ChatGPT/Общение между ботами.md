async def run_debate_round(user_message: str, round_num: int) -> tuple[str, str, bool]:

    """

    Один раунд спора между Claude и GPT

    Возвращает: (ответ Claude, ответ GPT, нужно_продолжать)

    gpt_silent = gpt_response.strip() == SILENCE_MARKER

    

    should_continue = not (claude_silent and gpt_silent) and round_num < MAX_ROUNDS

    

    return claude_response, gpt_response, should_continue

        # Отправляем в Telegram

            conversation_history.append(f"Claude: {claude_resp}")

        if gpt_resp.strip() != SILENCE_MARKER:

            conversation_history.append(f"ChatGPT: {gpt_resp}")

        

        if not should_continue:

            await context.bot.send_message(chat_id=chat_id, text="✅ Спор завершён")

            break

        

        round_num += 1

        await asyncio.sleep(1)  # Небольшая задержка между раундами

async def with_retry(coro, retries=2, timeout_s=20):

    """Обёртка с retry для API-вызовов"""

    for attempt in range(retries + 1):

        try:

            async with asyncio.timeout(timeout_s):

                return await coro()

        except Exception as e:

            if attempt == retries:

                raise

            print(f"⚠️ Попытка {attempt + 1} не удалась: {e}")

            await asyncio.sleep(1 + attempt * 2)

    async def _call():

            message = await loop.run_in_executor(

                None,

                    model="claude-sonnet-4-20250514",

                )

            )

            return extract_claude_text(message)

        except Exception as e:

            return f"[Ошибка Claude: {e}]"

    

    return await with_retry(_call)

                temperature=0.7

            choice = resp.choices[0]

            if isinstance(choice, dict):

                content = choice.get("message", {}).get("content")

            else:

                content = getattr(choice.message, "content", None)

            return (content or str(choice)).strip()

        except Exception as e:

    return await with_retry(_call)
